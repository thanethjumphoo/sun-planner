+    // Find category node

+
+    // Find process node under category

+      const children = allNodes.filter(n => n.parentId === parentId);
+      for (const child of children) {
+        nodeIds.push(child.id);
+        collectTree(child.id);
+      }
+    };
+
+    for (const p of processNodes) {
+      nodeIds.push(p.id);

+    }

+      })
